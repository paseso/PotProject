using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class Player : MonoBehaviour {

    private bool _isOnBrother = false;
    private bool _isHitBrother = false;
    private Transform _canvasTransform;
    private Transform _brotherTransform;

	// Use this for initialization
	void Start () {
		
	}
	
	// Update is called once per frame
	void Update () {
        if (Input.GetKey(KeyCode.RightArrow))
        {
            var pos = GetComponent<RectTransform>().localPosition;
            pos.x += 5f;
            GetComponent<RectTransform>().localPosition = pos;
        }
        else if (Input.GetKey(KeyCode.LeftArrow))
        {
            var pos = GetComponent<RectTransform>().localPosition;
            pos.x -= 5f;
            GetComponent<RectTransform>().localPosition = pos;
        }
        if(Input.GetKeyDown(KeyCode.Space) && _isHitBrother)
        {
            if (!_isOnBrother)
            {
                var pos = GetComponent<RectTransform>().localPosition;
                pos.y += 110;
                _brotherTransform.gameObject.GetComponent<RectTransform>().localPosition = pos;
                _brotherTransform.SetParent(transform);
                _isOnBrother = true;
            }
            else
            {
                _brotherTransform.SetParent(_canvasTransform);
                var pos = _brotherTransform.gameObject.GetComponent<RectTransform>().localPosition;
                pos.x += 150;
//                pos.y -= 100;
                _brotherTransform.gameObject.GetComponent<RectTransform>().localPosition = pos;
                _isOnBrother = false;
                _isHitBrother = false;
            }
        }

    }

    public void OnCollisionEnter2D(Collision2D collision)
    {
        if(collision.gameObject.name == "Brother" && !_isHitBrother)
        {
            _brotherTransform = collision.transform;
            _canvasTransform = _brotherTransform.parent;
            _isHitBrother = true;
        }
    }
}
